ltm rule oed_protal_redirect_6 {
when HTTP_REQUEST {
	set browser_url [string tolower [HTTP::host][HTTP::uri]/]

	if { [class match $browser_url contains datagroup_uri_list_lc] }  { 

        set part_of_url_matched [class match -name $browser_url contains datagroup_uri_list_lc]
		set rest_of_uri [string map -nocase [list $part_of_url_matched /] $browser_url]
		set redirect_site_from_datagroup [class match -value $browser_url contains datagroup_uri_list_lc]

		set redirect_value "$redirect_site_from_datagroup$rest_of_uri"

		if {$redirect_value ends_with "/"} {
			set rvlengthnew [expr {[string length $redirect_value] -2}] 
			set redirect_value [string range $redirect_value 0 $rvlengthnew]
		}
		
		HTTP::redirect "http://$redirect_value"
	}
}
}
